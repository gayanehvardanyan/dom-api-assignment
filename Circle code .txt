<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script>
<script>

var alpha = 0;

function Rotate() {
    alpha += 0.05;

    var r = 200;
    var xcenter = 200;
    var ycenter = 200;
    var newLeft = Math.floor(xcenter + (r * Math.cos(alpha)));
    var newTop = Math.floor(ycenter + (r * Math.sin(alpha)));
    $('#smallCircle').animate({
        top: newTop,
        left: newLeft,
    }, 1, function() {
        Rotate();
    });
}

function Stop()
{
	$('#smallCircle').stop(true, false);
}

</script>
